STM32 物联网天气站

基于 FreeRTOS + ESP8266 + SSD1306 的嵌入式天气预报系统

一、项目概述

1.1 项目目标

本项目基于 STM32F103C8T6（Blue Pill）微控制器，通过 ESP8266 WiFi 模块连接互联网，从心知天气 API 获取实时天气数据，在 0.96 寸 SSD1306 OLED 显示屏上展示天气信息，并支持按键切换页面显示时间。

三、FreeRTOS 任务分析

任务名优先级栈大小功能
WIFITaskNormal384×4=1536BESP8266 初始化 → WiFi 连接 → NTP 同步 → 每 30 分钟 HTTP GET

3.1 任务优先级设计意图

WIFitask 优先级最高（Normal），因为它负责网络通信，需要及时响应 ESP8266 数据避免缓冲区溢出。其他 4 个任务均为 Low 优先级，因为它们可以容忍一定延迟。

3.2 关键设计决策

  • 信号量初始值 0（不是 1）：确保 WeatherTask 在 WIFITask 首次产生数据前不会尝试解析空缓冲区
  • OledTask 统一管理显示：WeatherTask 只负责解析数据设置标志，不直接操作 OLED
  • 事件组分阶段同步：WiFi 连接 → HTTP GET 完成 → NTP 开始，三个步骤依次执行避免 USART1 竞争

五、FreeRTOS 调度 (freertos.c)

Core/Src/freertos.c 是项目的核心调度文件，包含全部 5 个任务的实现。

5.1 WIFITask — 网络通信任务

void StartWIFITask(void *argument)
{
    // 阶段1: ESP8266 硬件初始化
    while (esp8266_init() != 0) {
        // 显示"ESP8266 init fail"
        // 重试5次后卡死
    }

    // 阶段2: WiFi 连接
    while (esp8266_connect_wifi("YOUR_WIFI_SSID", "YOUR_WIFI_PASSWORD") != 0) {
        // 重试10次后卡死
    }

    // 阶段3: 通知 TimeTask 可以开始 NTP
    osEventFlagsSet(systemEventGroupHandle, WIFI_CONNECTED_BIT);

    // 阶段4: 本任务内完成 NTP 同步（不依赖 TimeTask）
    while (ntp_get_time("pool.ntp.org", &ts) != 0) osDelay(1000);
    system_time = ts + 8*3600;  // UTC+8
    time_synced = 1;

    // 阶段5: 主循环 — 每30分钟获取天气
    for (;;) {
        esp8266_http_get("api.seniverse.com", 80, path,
                         http_response, sizeof(http_response));
        if (成功) osSemaphoreRelease(weatherSemHandle);
        else /* 显示错误码 */;
        osDelay(1800000);  // 30分钟
    }
}

设计要点

  • NTP 同步在 WIFITask 中完成（不在 TimeTask），减少跨任务通信和时序依赖
  • ESP8266 初始化失败后重试 5 次，WiFi 连接失败后重试 10 次，均卡死防止空跑
  • 所有 OLED 操作前后均使用 osMutexAcquire/Release 保护

5.2 WeatherTask — 天气解析任务

void StartWeatherTask(void *argument)
{
    for (;;) {
        // 等待信号量，超时5秒
        if (osSemaphoreAcquire(weatherSemHandle, 5000) == osOK) {
            int ret = weather_parse_now(http_response, &parsed);
            if (ret == 0) {
                current_weather = parsed;   // 结构体拷贝
                weather_updated = 1;        // volatile 标志
            }
        }
        osDelay(100);
    }
}

关键设计

  • http_response 是全局缓冲区，WIFITask 写入后 WeatherTask 读取 — 通过信号量保证"写后读"顺序
  • weather_updated 使用 volatile 修饰，确保 OledTask 能正确读到最新值
  • 超时 5 秒防止信号量意外丢失时任务永久阻塞

5.4 KeyTask — 按键检测

void StartKeyTask(void *argument)
{
    for (;;) {

            osDelay(50);  // 消抖
            if (HAL_GPIO_ReadPin(GPIOB, GPIO_PIN_8) == GPIO_PIN_RESET) {

说明

六、ESP8266 驱动详解

Core/Src/esp8266.c + Core/Inc/esp8266.h

6.1 环形缓冲区 — 中断接收

static char esp8266_ring_buf[512];
static volatile uint16_t ring_head = 0;
static uint16_t ring_tail = 0;

// 中断回调: ISR 写入 (生产者)
void esp8266_rx_callback(uint8_t data)
{
    uint16_t next = (ring_head + 1) % ESP8266_RING_BUF_SIZE;
    if (next != ring_tail)          // 检查环形缓冲区是否满
        esp8266_ring_buf[ring_head++] = data;
}

// 任务读取 (消费者) — 支持批量读取
int esp8266_rx_read(char *buf, uint32_t len)
{
    int i = 0;
    while (i < len && ring_tail != ring_head) {
        buf[i++] = esp8266_ring_buf[ring_tail++];
        ring_tail %= ESP8266_RING_BUF_SIZE;
    }
    return i;
}

关键改进

  • 512 字节环形缓冲区：解决 +IPD 突发数据 410+ 字节的溢出问题
  • 批量读取：每轮最多读 64 字节（而非逐字节），读取速率 64 字节/ms 远大于 115200 baud 的 11.5 字节/ms
  • 无数据时 osDelay(1)：避免忙等待浪费 CPU

6.2 HTTP GET 实现

int esp8266_http_get("api.seniverse.com", 80, path, response, resp_len)
{
    // 1. AT+CIPSTART="TCP","api.seniverse.com",80
    // 2. 等待 OK (10s 超时)
    // 3. AT+CIPSEND=长度
    // 4. 等待 ">" 提示符 (5s 超时)
    // 5. 发送 HTTP GET 请求
    // 6. 批量读取 15s, 每轮最多 64 字节
    // 7. 事后数据清理:
    //    a) 找第一个 '{' 或 '['
    //    b) 原地过滤 +IPD,数字: 帧头
    //    c) memmove 到缓冲区开头
}

数据清理算法（核心）

// 跳过 +IPD,N: 帧头的原地过滤算法
char *r = json, *w = json;
while (*r) {
    if (strncmp(r, "+IPD,", 5) == 0) {
        r += 5;
        while (*r >= '0' && *r <= '9') r++;  // 跳过长数字
        if (*r == ':') r++;  // 跳过冒号
        continue;  // 丢弃整个 +IPD,N: 帧头
    }
    *w++ = *r++;
}
*w = '\0';
memmove(response, json, w - json + 1);

算法原理

  • r（读指针）始终 >= w（写指针），不会互相覆盖
  • 遇到 "+IPD," 后跳过数字和冒号，不复制到输出
  • 其余字符正常复制
  • 最后 memmove 将过滤后的数据移到缓冲区头部

七、天气数据解析 (weather.c)

Core/Src/weather.c + Core/Inc/weather.h

7.1 WeatherData_t 结构体

typedef struct {
    char city[32];      // 城市名 (如 "Hangzhou")
    char weather[32];   // 天气描述 (如 "Cloudy")
    char temp[8];       // 温度 (如 "27")
    char humidity[8];   // 湿度 (始终 "--", 免费版不提供)
} WeatherData_t;

7.2 JSON 解析函数 — weather_parse_now

int weather_parse_now(const char* json_response, WeatherData_t* weather)
{
    // 空指针检查
    if (!json_response || !weather) return -1;

    // 定位 JSON 结构: results → [...] → {...}
    results_start = strstr(json, "results");     // 搜索 "results" 键
    first_brace  = strchr(results_start, '[');   // 找到数组开始 [
    first_obj    = strchr(first_brace, '{');     // 找到第一个对象 {

    // 解析字段
    if (location_start)
        json_find_value(location_start, "name", weather->city, ...);
    if (now_start) {
        json_find_value(now_start, "text", weather->weather, ...);
        json_find_value(now_start, "temperature", weather->temp, ...);
        json_find_value(now_start, "humidity", weather->humidity, ...);
    }

    // 过滤非 ASCII 字符 + 空字符串 Fallback
    sanitize_to_ascii(...);
    if (strlen(city) == 0) strcpy(city, "Unknown");
    if (strlen(temp) == 0) strcpy(temp, "--");
}

7.3 json_find_value 实现细节

static char* json_find_value(const char* json, const char* key, char* value, int value_len)
{
    p = strstr(json, key);          // 找键名
    p = strchr(p, ':');            // 跳过冒号
    p++;  while (*p == ' ') p++;    // 跳过空格

    if (*p == '"') {               // 字符串值: "value"
        p++;  end = strchr(p, '"');
        strncpy(value, p, end - p);
    } else {                       // 数字值: 123
        while (*end != ',' && *end != '}') end++;
        strncpy(value, p, end - p);
    }
}

解析流程

HTTP 响应→ 找 "results"→ 找 '['→ 找 '{'→ 找 "location" → name→ 找 "now" → text/temperature/humidity→ ASCII 过滤 + Fallback

7.4 防御性编程

  • 每个 strstr/strchr 返回值都检查 NULL，返回不同的错误码 (-1~-4)
  • sanitize_to_ascii 过滤非 ASCII 字符（中文字符 UTF-8 多字节）
  • 所有字段均有默认值 Fallback（"Unknown" / "--"）
  • 从修改报告可知，曾发生过 first_brace 为 NULL 导致的 HardFault

八、OLED 显示驱动 (oled.c / oled.h)

8.1 I2C 通信基础

// oled.h 关键宏定义
#define OLED_I2C_ADDR    0x78    // SSD1306 I2C 地址 (7bit 左移1位)
#define OLED_CMD         0       // 控制字节: 下一个字节是命令
#define OLED_DATA        0x40    // 控制字节: 下一个字节是数据

// I2C 写字节函数
static void oled_wr_byte(uint8_t data, uint8_t cmd)
{
    uint8_t buf[2];
    buf[0] = cmd ? OLED_DATA : OLED_CMD;  // cmd=0→0x00, cmd=1→0x40
    buf[1] = data;
    HAL_I2C_Master_Transmit(&hi2c1, OLED_I2C_ADDR, buf, 2, 100);
}

重要：OLED_DATA 为什么是 0x40？

SSD1306 驱动芯片的 I2C 协议中，每个数据包的第一个字节是控制字节。bit6 = D/C# 位：0 表示后续是命令，1 表示后续是数据。因此数据控制字节必须是 0x40（bit6=1），而不是简单的逻辑值 1。早期版本将此宏误定义为 1，导致所有显示数据被芯片解释为命令，产生乱码。

8.2 双缓冲显示

static uint8_t oled_gram[128][8];  // 1KB 显存: 128列 × 8页

// 只清缓冲区 — 解决闪烁问题
void oled_clear_buffer(void)
{
    for (i=0; i<8; i++)
        for (n=0; n<128; n++)
            oled_gram[n][i] = 0x00;
}

// 将整个 GRAM 刷到 OLED
void oled_refresh_gram(void)
{
    for (i=0; i<8; i++) {
        oled_wr_byte(0xb0+i, OLED_CMD);  // 设置页地址
        oled_wr_byte(0x00, OLED_CMD);    // 列低4位
        oled_wr_byte(0x10, OLED_CMD);    // 列高4位
        for (n=0; n<128; n++)
            oled_wr_byte(oled_gram[n][i], OLED_DATA);  // 128字节/页
    }
}

双缓冲策略

  • 先在内存中的 oled_gram 绘制
  • 绘制完成调用 oled_refresh_gram 一次性送显
  • 清屏时只清缓冲区不清显存（oled_clear_buffer），避免了屏幕闪烁

8.3 大字显示函数（2x 缩放）

void oled_show_big_char(uint8_t x, uint8_t y, char c)
{
    uint8_t *pfont = (uint8_t *)oled_asc2_1608[c - ' '];

    for (t = 0; t < 16; t++) {
        temp = pfont[t];
        for (t1 = 0; t1 < 8; t1++) {
            if (temp & 0x80) {
                // 每个像素点放大为 2×2 方块
                oled_draw_point(x, y, 1);
                oled_draw_point(x+1, y, 1);
                oled_draw_point(x, y+1, 1);
                oled_draw_point(x+1, y+1, 1);
            }
            temp <<= 1;
            y += 2;  // 垂直步长 2
        }
        x += 2;  // 水平步长 2
    }
}

原理

  • 从标准 8×16 字库 oled_asc2_1608 读取字符像素
  • 每个原始像素点扩展为 2×2 像素块
  • 最终每个大字占 16×32 像素，一行可显示 8 个大字

8.4 字库风格

使用正点原子 ATK-OLED 标准 8×16 ASCII 字库（95 个字符：0x20~0x7E），包含完整的字母大小写、数字和符号。

九、NTP 时间同步 (ntp.c)

9.1 NTP 协议实现

int ntp_get_time("pool.ntp.org", ×tamp)
{
    // 1. UDP 连接: AT+CIPSTART="UDP","pool.ntp.org",123,123,0
    // 2. AT+CIPSEND=48
    // 3. 发送 48 字节 NTP 请求包 (首字节 0x1B, 其余 0)
    // 4. 批量读取 +IPD,48: 响应 (10s 超时)
    // 5. 解析 +IPD,48: 帧头, 提取 byte 40-43

    // 大端 4 字节 → NTP 时间戳
    uint32_t ntp_time = (p[40] << 24) | (p[41] << 16) |
                        (p[42] << 8)  |  p[43];

    // NTP → Unix: 减去 1900→1970 的秒差
    *timestamp = ntp_time - 2208988800UL;
}

9.2 NTP 协议细节

字段偏移说明
LI/VN/Modebyte 00x1B = VN=3 (版本3) + Mode=3 (客户端)
Stratumbyte 1请求包填 0
Poll/Precbyte 2-3请求包填 0
保留字段byte 4-39全 0
Transmit Timestampbyte 40-47服务器填写, 40-43 是秒部分（大端）

9.3 时间格式化

void ntp_format_time(char* buf, uint32_t timestamp)
{
    uint32_t sec = timestamp % 86400;      // 一天秒数
    h = sec / 3600;  m = (sec % 3600) / 60;  s = sec % 60;
    sprintf(buf, "%02d:%02d:%02d", h, m, s);  // HH:MM:SS
}

注意事项

  • NTP 请求包含 0x00 字节（二进制数据），必须用 esp8266_send_raw 而非 esp8266_send_at（strlen 会在第一个 0x00 截断）
  • 时区固定 UTC+8（杭州），在 WIFITask 中加 8 小时
  • NTP 使用 UDP 协议比 HTTP 获取时间更高效（48 字节 vs 数百字节）

十、Bug 修复记录（重点）

10.1 OLED 显示乱码 — OLED_DATA 宏定义错误

I2C 协议SSD1306

根因：#define OLED_DATA 1，正确的 SSD1306 I2C 数据控制字节应为 0x40。

修复：改为 #define OLED_DATA 0x40（oled.h:13）

10.2 RAM 溢出 — .bss 段超 20KB

内存优化FreeRTOS

根因：FreeRTOS heap_4 的 ucHeap[configTOTAL_HEAP_SIZE] 在 .bss 段中，之前设为 16KB 导致 .bss 超 20KB。

修复：

  • configTOTAL_HEAP_SIZE: 16KB → 8KB
  • http_response: 2048 → 1024
  • ESP8266 环形缓冲区: 512 → 256（后又调回 512）
  • 各任务栈缩减

10.3 HardFault — first_brace 空指针

空指针JSON 解析

根因：strchr(results_start, '[') 返回 NULL 后未检查，直接传给下一个 strchr。

修复：添加 NULL 检查并返回不同错误码（weather.c:76-85）

// 改前: 直接使用 first_brace 不检查
const char* first_brace = strchr(results_start, '[');
const char* first_obj = strchr(first_brace, '{');  // ← CRASH if NULL

// 改后: 逐级检查
if (first_brace == NULL) return -3;
const char* first_obj = strchr(first_brace, '{');
if (first_obj == NULL) return -4;

10.4 USART1 竞争 — WIFITask 与 TimeTask 冲突

RTOS 同步事件组

根因：WIFITask 的 HTTP GET 和 TimeTask 的 NTP 同时向 ESP8266 发 AT 命令，数据在 USART1 上交错，ESP8266 收到乱码返回 "busy"。

修复：使用事件组分阶段同步（freertos.c）

// TimeTask 分两步等待
osEventFlagsWait(systemEventGroupHandle, WIFI_CONNECTED_BIT, ...);  // 等WiFi
osEventFlagsWait(systemEventGroupHandle, NTP_ALLOWED_BIT, ...);    ">// 等HTTP完成

// WIFITask HTTP GET 完成后释放
osEventFlagsSet(systemEventGroupHandle, NTP_ALLOWED_BIT);

10.5 环形缓冲区溢出 — +IPD 数据截断

缓冲区突发数据

根因：ESP8266 以 115200 baud 突发 410+ 字节，但逐字节读取仅 1 字节/ms 远低于到达速率 11.5 字节/ms，256 字节缓冲区迅速填满丢包。

修复：

  • 环形缓冲区 256 → 512
  • 批量读取：每轮最多 64 字节，读取速率 64 > 11.5 字节/ms
  • 无数据时才 osDelay(1)

10.6 +IPD 分帧污染 JSON 数据

数据后处理ESP8266

根因：HTTP 大响应被 ESP8266 拆成多个 +IPD,N: 帧，前一帧 JSON 末尾和后一帧 "+IPD,N:" 帧头混在缓冲区中。

修复：先收集全部原始数据，再通过原地扫描过滤 +IPD,N: 帧头。

10.7 OLED 闪烁 — 清屏送显策略

优化双缓冲

根因：oled_clear() 清缓冲区后立即送显（屏幕变白），再画新内容二次送显。

修复：新增 oled_clear_buffer() 只清缓冲区不送显，绘制完成后一次 oled_refresh_gram()。
