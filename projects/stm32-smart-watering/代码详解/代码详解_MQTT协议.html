MQTT 协议栈模块详解

MqttKit v1.6 — 文件：Onenet/MqttKit.c + Onenet/MqttKit.h  |  作者：Li Zhi Qu

1. 模块概述

MqttKit 是由 Li Zhi Qu 编写的轻量级 MQTT v3.1.1 协议栈实现，专为资源受限的嵌入式平台（STM32 + ESP8266）设计。该模块实现了 MQTT 3.1.1 协议的核心子集：

  • CONNECT — 连接建立（含用户名/密码鉴权）
  • CONNACK — 连接确认解析
  • SUBSCRIBE — 主题订阅
  • PUBLISH (QoS 0/1) — 消息发布与接收
  • PUBACK — QoS 1 发布确认
  • PINGREQ / PINGRESP — 心跳保活

缓冲区采用 malloc / free 动态管理，同时支持静态缓冲区模式，灵活适配不同内存策略。

协议版本：MQTT v3.1.1（协议级别 4） |  支持 QoS：0（至多一次）、1（至少一次）

2. 数据结构

核心数据结构 MQTT_PACKET_STRUCTURE 用于描述一个 MQTT 协议缓冲区：

typedef struct {
  uint8_t   *_data;       // 协议缓冲区指针
  uint16_t   _len;        // 已写入的数据长度
  uint16_t   _size;       // 缓冲区总容量
  uint8_t    _memFlag;    // 内存标记
} MQTT_PACKET_STRUCTURE;

_memFlag 取值为以下枚举常量之一：

enum { MEM_FLAG_NULL = 0, // 无内存分配 MEM_FLAG_ALLOC = 1, // malloc 动态分配 MEM_FLAG_STATIC = 2, // 静态缓冲区 };
设计考量：区分 STATIC 与 ALLOC 标记使得同一套 API 可以同时服务于栈上静态数组和堆上动态内存，避免不必要的内存拷贝。

3. 缓冲区管理

MQTT_NewBuffer — 创建缓冲区

MQTT_PACKET_STRUCTURE MQTT_NewBuffer(uint16_t size)
{
  MQTT_PACKET_STRUCTURE packet;
  packet._data = NULL;
  packet._len  = 0;
  packet._size = 0;
  packet._memFlag = MEM_FLAG_NULL;

  if (size == 0) return packet;

  packet._data = (uint8_t *)malloc(size);
  if (packet._data) {
    packet._size = size;
    packet._memFlag = MEM_FLAG_ALLOC;
  }
  return packet;
}

MQTT_DeleteBuffer — 释放缓冲区

void MQTT_DeleteBuffer(MQTT_PACKET_STRUCTURE *packet)
{
  if (packet && packet->_data && packet->_memFlag == MEM_FLAG_ALLOC) {
    free(packet->_data);
  }
  packet->_data = NULL;
  packet->_len  = 0;
  packet->_size = 0;
  packet->_memFlag = MEM_FLAG_NULL;
}
注意：只有 MEM_FLAG_ALLOC 标记的缓冲区才会被 free 释放。静态缓冲区（MEM_FLAG_STATIC）由调用方自行管理生命周期。

静态缓冲区使用示例：

uint8_t stackBuf[128];
MQTT_PACKET_STRUCTURE pkt;
pkt._data    = stackBuf;
pkt._size    = sizeof(stackBuf);
pkt._len     = 0;
pkt._memFlag = MEM_FLAG_STATIC;

4. 剩余长度编码（Variable-Length Encoding）

MQTT v3.1.1 协议使用可变长度编码（Variable Length Encoding）表示剩余长度（Remaining Length）。每个字节的低 7 位存放数据，最高位（bit 7）作为延续标志位：

  • 若最高位为 1，表示后续还有字节
  • 若最高位为 0，表示当前是最后一个字节
  • 最大支持 268,435,455 字节（约 256 MB），最多使用 4 个字节

编码：MQTT_DumpLength

uint8_t MQTT_DumpLength(uint32_t len, uint8_t *buf)
{
  uint8_t bytes = 0;
  do {
    uint8_t byte = len % 128;
    len /= 128;
    if (len > 0) byte |= 0x80;
    buf[bytes++] = byte;
  } while (len > 0);
  return bytes;
}

解码：MQTT_ReadLength

uint32_t MQTT_ReadLength(uint8_t *buf, uint8_t *count)
{
  uint32_t len = 0;
  uint8_t  multiplier = 1;
  uint8_t  byte;
  *count = 0;

  do {
    byte = buf[(*count)++];
    len += (byte & 0x7F) * multiplier;
    multiplier *= 128;
  } while (byte & 0x80);

  return len;
}
示例：长度 64（0x40）→ 编码为 [0x40]（1 字节）；长度 321（0x141）→ 编码为 [0x81, 0x02]（2 字节）。

5. 数据包类型识别

MQTT_UnPacketRecv 是解析入口函数。MQTT 固定报头（Fixed Header）的第一个字节的高 4 位表示数据包类型：

uint8_t MQTT_UnPacketRecv(uint8_t *buf, uint8_t len)
{
  uint8_t type = (buf[0] >> 4) & 0x0F;   // 取高4位 = 包类型

  switch (type) {
    case MQTT_PUBLISH: {
      uint8_t topicLen = (buf[2] << 8) | buf[3];
      // 检查主题是否以 $creq 开头 => 命令下发
      if (topicLen > 5 && memcmp(buf + 4, "$creq", 5) == 0) {
        return MQTT_CMD_TYPE;
      }
      return MQTT_PUBLISH;
    }
    case MQTT_CONNACK:  return MQTT_CONNACK;
    case MQTT_SUBACK:   return MQTT_SUBACK;
    case MQTT_PUBACK:   return MQTT_PUBACK;
    case MQTT_PINGRESP: return MQTT_PINGRESP;
    default:           return MQTT_UNKNOW_TYPE;
  }
}
特殊处理：对于 PUBLISH 类型，额外检查主题（Topic）中是否包含 $creq 前缀。OneNet 平台的下行命令使用此特殊主题，被识别为 MQTT_CMD_TYPE 单独处理。

6. CONNECT 数据包

MQTT_PacketConnect 构造连接报文，包含协议名、协议级别、标志位、保活时间和负载信息：

void MQTT_PacketConnect(MQTT_PACKET_STRUCTURE *pkt,
  char *clientID, char *username, char *password,
  uint16_t keepalive, uint8_t cleanSession)
{
  uint8_t buf[256];
  uint16_t idx = 0;

  // 协议名 "MQTT"
  buf[idx++] = 0x00; buf[idx++] = 0x04;
  buf[idx++] = 'M';  buf[idx++] = 'Q';
  buf[idx++] = 'T';  buf[idx++] = 'T';

  // 协议级别 4 = MQTT v3.1.1
  buf[idx++] = 0x04;

  // 连接标志位：CleanSession, Will, Username, Password
  uint8_t flags = 0;
  if (cleanSession) flags |= 0x02;
  if (username)     flags |= 0x80;
  if (password)     flags |= 0x40;
  buf[idx++] = flags;

  // Keep Alive (秒)
  buf[idx++] = (keepalive >> 8) & 0xFF;
  buf[idx++] = keepalive & 0xFF;

  // Client ID (UTF-8 编码)
  uint16_t clen = strlen(clientID);
  buf[idx++] = (clen >> 8) & 0xFF;
  buf[idx++] = clen & 0xFF;
  memcpy(buf + idx, clientID, clen); idx += clen;

  // Username
  if (username) {
    uint16_t ulen = strlen(username);
    buf[idx++] = (ulen >> 8) & 0xFF;
    buf[idx++] = ulen & 0xFF;
    memcpy(buf + idx, username, ulen); idx += ulen;
  }

  // Password
  if (password) {
    uint16_t plen = strlen(password);
    buf[idx++] = (plen >> 8) & 0xFF;
    buf[idx++] = plen & 0xFF;
    memcpy(buf + idx, password, plen); idx += plen;
  }

  // 剩余长度 + 固定报头
  uint8_t lenBuf[4];
  uint8_t lenBytes = MQTT_DumpLength(idx, lenBuf);

  pkt->_data[0] = (MQTT_CONNECT << 4) | 0x00;   // 固定报头第1字节
  memcpy(pkt->_data + 1, lenBuf, lenBytes);
  memcpy(pkt->_data + 1 + lenBytes, buf, idx);
  pkt->_len = 1 + lenBytes + idx;
}

在智能灌溉系统中，OneNet 平台的鉴权信息映射如下：

MQTT 字段OneNet 映射说明
Client IDDEVID设备 ID（唯一标识）
UsernamePROID产品 ID
PasswordTOKEN设备密钥（鉴权令牌）

7. CONNACK 解析

MQTT_UnPacketConnectAck 解析服务器返回的连接确认报文。CONNACK 固定为 4 字节：

  • 字节 0：固定报头（0x20）
  • 字节 1：剩余长度（0x02）
  • 字节 2：Session Present 标志
  • 字节 3：连接返回码
uint8_t MQTT_UnPacketConnectAck(uint8_t *buf)
{
  // buf[2]: Session Present (bit 0)
  // buf[3]: Return Code (0 = 连接接受)
  return buf[3];   // 返回 0 表示成功
}
返回码含义：0 = 连接接受，1 = 协议版本拒绝，2 = 标识符被拒，3 = 服务器不可用，4 = 用户名或密码错误，5 = 未授权。

8. SUBSCRIBE 数据包

MQTT_PacketSubscribe 构造订阅请求报文。每个订阅主题后跟一个 QoS 字节：

uint8_t MQTT_PacketSubscribe(MQTT_PACKET_STRUCTURE *pkt,
  char *topic, uint8_t qos, uint16_t packetId)
{
  uint16_t topicLen = strlen(topic);
  uint8_t payload[256], idx = 0;

  // Packet Identifier
  payload[idx++] = (packetId >> 8) & 0xFF;
  payload[idx++] = packetId & 0xFF;

  // Topic Filter (UTF-8 编码)
  payload[idx++] = (topicLen >> 8) & 0xFF;
  payload[idx++] = topicLen & 0xFF;
  memcpy(payload + idx, topic, topicLen); idx += topicLen;

  // QoS Byte
  payload[idx++] = qos & 0x03;

  // 剩余长度编码
  uint8_t lenBuf[4];
  uint8_t lenBytes = MQTT_DumpLength(idx, lenBuf);

  pkt->_data[0] = (MQTT_SUBSCRIBE << 4) | 0x02;  // bit 1 = 固定为 1 (SUBSCRIBE)
  memcpy(pkt->_data + 1, lenBuf, lenBytes);
  memcpy(pkt->_data + 1 + lenBytes, payload, idx);
  pkt->_len = 1 + lenBytes + idx;

  return 1;
}
偏移大小描述
01固定报头：0x82（SUBSCRIBE 类型 + bit 1 置位）
1+可变剩余长度（可变长编码）
...2Packet Identifier（报文标识符）
...2主题过滤器长度（MSB + LSB）
...topicLen主题过滤器内容
...1QoS 字节（低 2 位有效）

9. PUBLISH 数据包

编码：MQTT_PacketPublish

void MQTT_PacketPublish(MQTT_PACKET_STRUCTURE *pkt,
  char *topic, uint8_t *payload, uint16_t payloadLen,
  uint8_t qos, uint16_t packetId)
{
  uint16_t topicLen = strlen(topic);
  uint8_t varHeader[256], idx = 0;

  // Topic Name (UTF-8)
  varHeader[idx++] = (topicLen >> 8) & 0xFF;
  varHeader[idx++] = topicLen & 0xFF;
  memcpy(varHeader + idx, topic, topicLen); idx += topicLen;

  // QoS 1/2 时需要 Packet ID
  if (qos > 0) {
    varHeader[idx++] = (packetId >> 8) & 0xFF;
    varHeader[idx++] = packetId & 0xFF;
  }

  // 剩余长度 = 变长报头 + 负载
  uint32_t remainLen = idx + payloadLen;
  uint8_t lenBuf[4];
  uint8_t lenBytes = MQTT_DumpLength(remainLen, lenBuf);

  // 固定报头：flags = QoS << 1
  pkt->_data[0] = (MQTT_PUBLISH << 4) | (qos << 1);
  memcpy(pkt->_data + 1, lenBuf, lenBytes);
  memcpy(pkt->_data + 1 + lenBytes, varHeader, idx);
  memcpy(pkt->_data + 1 + lenBytes + idx, payload, payloadLen);
  pkt->_len = 1 + lenBytes + idx + payloadLen;
}

解码：MQTT_UnPacketPublish

void MQTT_UnPacketPublish(uint8_t *buf,
  char **topic, uint8_t **payload, uint16_t *payloadLen,
  uint8_t *qos, uint16_t *packetId)
{
  uint8_t count;
  uint32_t remainLen = MQTT_ReadLength(buf + 1, &count);
  uint16_t idx = 1 + count;

  // 提取 QoS (从固定报头 bit 1-2)
  *qos = (buf[0] >> 1) & 0x03;

  // 解析 Topic
  uint16_t topicLen = (buf[idx] << 8) | buf[idx + 1];
  idx += 2;
  *topic = (char *)(buf + idx);
  idx += topicLen;

  // QoS 1/2 时提取 Packet Identifier
  if (*qos > 0) {
    *packetId = (buf[idx] << 8) | buf[idx + 1];
    idx += 2;
  }

  // 负载
  *payload    = buf + idx;
  *payloadLen = remainLen - (idx - (1 + count));
}
QoS 说明：对于 QoS 0，PUBLISH 报文中不包含 Packet Identifier。QoS 1 时设置 Packet ID 并期望接收方回复 PUBACK。QoS 2 未在此实现中使用。

10. PUBACK 数据包

PUBACK 用于对 QoS 1 的 PUBLISH 消息进行确认。报文结构非常简单：

void MQTT_PacketPublishAck(MQTT_PACKET_STRUCTURE *pkt,
  uint16_t packetId)
{
  pkt->_data[0] = (MQTT_PUBACK << 4) | 0x00;  // 0x40
  pkt->_data[1] = 0x02;                         // 剩余长度 = 2
  pkt->_data[2] = (packetId >> 8) & 0xFF;       // Packet ID MSB
  pkt->_data[3] = packetId & 0xFF;              // Packet ID LSB
  pkt->_len = 4;
}
字节值说明
00x40固定报头：PUBACK 类型
10x02剩余长度（固定为 2）
2Packet ID MSB报文标识符高字节
3Packet ID LSB报文标识符低字节

11. PING 数据包

心跳请求（PINGREQ）是最简单的 MQTT 报文：

void MQTT_PacketPing(MQTT_PACKET_STRUCTURE *pkt)
{
  pkt->_data[0] = (MQTT_PINGREQ << 4) | 0x00;  // 0xC0
  pkt->_data[1] = 0x00;                         // 剩余长度 = 0
  pkt->_len = 2;
}
心跳间隔：主循环中每 30 秒（3000 个主循环周期）调用一次 MQTT_PacketPing 并通过 ESP8266 发送。服务器回复 PINGRESP（0xD0 0x00）表示连接仍存活。

MQTT 数据包类型枚举定义在 MqttKit.h：

MqttKit.c（协议实现核心）

#include "MqttKit.h"

MQTT_PACKET_STRUCTURE MQTT_NewBuffer(uint16_t size)
{
  MQTT_PACKET_STRUCTURE pkt = {NULL, 0, 0, MEM_FLAG_NULL};
  if (size == 0) return pkt;
  pkt._data = (uint8_t *)malloc(size);
  if (pkt._data) { pkt._size = size; pkt._memFlag = MEM_FLAG_ALLOC; }
  return pkt;
}

void MQTT_DeleteBuffer(MQTT_PACKET_STRUCTURE *pkt)
{
  if (pkt && pkt->_data && pkt->_memFlag == MEM_FLAG_ALLOC)
    free(pkt->_data);
  pkt->_data = NULL; pkt->_len = 0; pkt->_size = 0; pkt->_memFlag = MEM_FLAG_NULL;
}

uint8_t MQTT_DumpLength(uint32_t len, uint8_t *buf)
{
  uint8_t cnt = 0;
  do {
    uint8_t d = len % 128; len /= 128;
    if (len > 0) d |= 0x80;
    buf[cnt++] = d;
  } while (len > 0);
  return cnt;
}

uint32_t MQTT_ReadLength(uint8_t *buf, uint8_t *cnt)
{
  uint32_t len = 0, mul = 1; *cnt = 0;
  uint8_t b;
  do {
    b = buf[(*cnt)++];
    len += (b & 0x7F) * mul;
    mul *= 128;
  } while (b & 0x80);
  return len;
}

uint8_t MQTT_UnPacketRecv(uint8_t *buf, uint8_t len)
{
  uint8_t type = (buf[0] >> 4) & 0x0F;
  if (type == MQTT_PUBLISH) {
    uint16_t tlen = (buf[2] << 8) | buf[3];
    if (tlen > 5 && memcmp(buf + 4, "$creq", 5) == 0)
      return MQTT_CMD_TYPE;
    return MQTT_PUBLISH;
  }
  switch (type) {
    case MQTT_CONNACK:   return MQTT_CONNACK;
    case MQTT_SUBACK:    return MQTT_SUBACK;
    case MQTT_PUBACK:    return MQTT_PUBACK;
    case MQTT_PINGRESP:  return MQTT_PINGRESP;
    default:            return MQTT_UNKNOW_TYPE;
  }
}

void MQTT_PacketConnect(MQTT_PACKET_STRUCTURE *pkt,
  char *cid, char *user, char *pwd,
  uint16_t ka, uint8_t clean)
{
  uint8_t b[256], idx = 0;
  b[idx++] = 0x00; b[idx++] = 0x04; // "MQTT" len
  b[idx++] = 'M'; b[idx++] = 'Q'; b[idx++] = 'T'; b[idx++] = 'T';
  b[idx++] = 0x04;  // 协议级别
  uint8_t f = clean ? 0x02 : 0;
  if (user) f |= 0x80; if (pwd) f |= 0x40;
  b[idx++] = f;
  b[idx++] = (ka >> 8) & 0xFF; b[idx++] = ka & 0xFF;
  uint16_t cl = strlen(cid);
  b[idx++] = (cl >> 8) & 0xFF; b[idx++] = cl & 0xFF;
  memcpy(b + idx, cid, cl); idx += cl;
  if (user) {
    uint16_t ul = strlen(user);
    b[idx++] = (ul >> 8) & 0xFF; b[idx++] = ul & 0xFF;
    memcpy(b + idx, user, ul); idx += ul;
  }
  if (pwd) {
    uint16_t pl = strlen(pwd);
    b[idx++] = (pl >> 8) & 0xFF; b[idx++] = pl & 0xFF;
    memcpy(b + idx, pwd, pl); idx += pl;
  }
  uint8_t lb[4], lc = MQTT_DumpLength(idx, lb);
  pkt->_data[0] = (MQTT_CONNECT << 4);
  memcpy(pkt->_data + 1, lb, lc);
  memcpy(pkt->_data + 1 + lc, b, idx);
  pkt->_len = 1 + lc + idx;
}

uint8_t MQTT_UnPacketConnectAck(uint8_t *buf) { return buf[3]; }

uint8_t MQTT_PacketSubscribe(MQTT_PACKET_STRUCTURE *pkt,
  char *topic, uint8_t qos, uint16_t pid)
{
  uint16_t tl = strlen(topic);
  uint8_t b[256], idx = 0;
  b[idx++] = (pid >> 8) & 0xFF; b[idx++] = pid & 0xFF;
  b[idx++] = (tl >> 8) & 0xFF;  b[idx++] = tl & 0xFF;
  memcpy(b + idx, topic, tl); idx += tl;
  b[idx++] = qos & 0x03;
  uint8_t lb[4], lc = MQTT_DumpLength(idx, lb);
  pkt->_data[0] = (MQTT_SUBSCRIBE << 4) | 0x02;
  memcpy(pkt->_data + 1, lb, lc);
  memcpy(pkt->_data + 1 + lc, b, idx);
  pkt->_len = 1 + lc + idx;
  return 1;
}

void MQTT_PacketPublish(MQTT_PACKET_STRUCTURE *pkt,
  char *topic, uint8_t *payload, uint16_t plen,
  uint8_t qos, uint16_t pid)
{
  uint16_t tl = strlen(topic), idx = 0;
  uint8_t vh[256];
  vh[idx++] = (tl >> 8) & 0xFF; vh[idx++] = tl & 0xFF;
  memcpy(vh + idx, topic, tl); idx += tl;
  if (qos > 0) {
    vh[idx++] = (pid >> 8) & 0xFF; vh[idx++] = pid & 0xFF;
  }
  uint32_t rml = idx + plen;
  uint8_t lb[4], lc = MQTT_DumpLength(rml, lb);
  pkt->_data[0] = (MQTT_PUBLISH << 4) | (qos << 1);
  memcpy(pkt->_data + 1, lb, lc);
  memcpy(pkt->_data + 1 + lc, vh, idx);
  memcpy(pkt->_data + 1 + lc + idx, payload, plen);
  pkt->_len = 1 + lc + idx + plen;
}

void MQTT_UnPacketPublish(uint8_t *buf,
  char **topic, uint8_t **payload, uint16_t *plen,
  uint8_t *qos, uint16_t *pid)
{
  uint8_t cnt; uint32_t rml = MQTT_ReadLength(buf + 1, &cnt);
  uint16_t idx = 1 + cnt;
  *qos = (buf[0] >> 1) & 0x03;
  uint16_t tl = (buf[idx] << 8) | buf[idx + 1]; idx += 2;
  *topic = (char *)(buf + idx); idx += tl;
  if (*qos > 0) { *pid = (buf[idx] << 8) | buf[idx + 1]; idx += 2; }
  *payload = buf + idx;
  *plen = rml - (idx - (1 + cnt));
}

void MQTT_PacketPublishAck(MQTT_PACKET_STRUCTURE *pkt, uint16_t pid)
{
  pkt->_data[0] = (MQTT_PUBACK << 4);
  pkt->_data[1] = 0x02;
  pkt->_data[2] = (pid >> 8) & 0xFF;
  pkt->_data[3] = pid & 0xFF;
  pkt->_len = 4;
}

void MQTT_PacketPing(MQTT_PACKET_STRUCTURE *pkt)
{
  pkt->_data[0] = (MQTT_PINGREQ << 4);  // 0xC0
  pkt->_data[1] = 0x00;
  pkt->_len = 2;
}
