ESP8266 WiFi 驱动模块详解

文件：Onenet/esp8266.c Onenet/esp8266.h  |  基于 STM32 智能浇花系统

1. 模块概述

本模块驱动 ESP8266 WiFi 模组，通过 AT 指令集 完成 Wi-Fi 联网和 TCP 连接，最终与 OneNet 云平台的 MQTT Broker 通信（mqtts.heclouds.com:1883）。

模块核心功能：

  • 使用 USART1 与 ESP8266 通信（115200 波特率）
  • 通过 AT 指令 配置 ESP8266 的工作模式、连接 Wi-Fi、建立 TCP 连接
  • 提供 非阻塞中断接收 机制接收 ESP8266 返回的数据
  • 支持 AT 指令响应等待（超时重试）、原始数据收发、+IPD 帧解析

依赖 BSP/bsp_usart.c 提供 USART 初始化和字节发送函数；BSP/bsp_delay.c 提供毫秒级延时。

2. 硬件连接

ESP8266 通过 USART1 与 STM32 全双工连接：

STM32 引脚ESP8266 引脚说明
PA9 (USART1_TX)RXDSTM32 发送 → ESP8266 接收
PA10 (USART1_RX)TXDESP8266 发送 → STM32 接收
3.3VVCC / CH_PD电源正极（需 3.3V / ≥300mA）
GNDGND共地
注意： ESP8266 的 CH_PD（使能）引脚必须拉高才能工作；RST 引脚可通过 10kΩ 上拉至 3.3V。电源需提供至少 300 mA 电流，建议加 100 μF 电解电容滤波。

USART1 在 bsp_usart.c 中初始化，关键配置如下：

// PA9 = TX (复用推挽), PA10 = RX (浮空输入)
gpioInitStruct.GPIO_Mode = GPIO_Mode_AF_PP;
gpioInitStruct.GPIO_Pin  = GPIO_Pin_9;    // TX
GPIO_Init(GPIOA, &gpioInitStruct);

gpioInitStruct.GPIO_Mode = GPIO_Mode_IN_FLOATING;
gpioInitStruct.GPIO_Pin  = GPIO_Pin_10;   // RX
GPIO_Init(GPIOA, &gpioInitStruct);

// 115200-8N1, 开启 RXNE 中断
usartInitStruct.USART_BaudRate = 115200;
USART_Init(USART1, &usartInitStruct);
USART_ITConfig(USART1, USART_IT_RXNE, ENABLE);

3. USART1 中断接收

ESP8266 返回的数据是异步到达的（AT 响应、+IPD 数据推送），因此采用 USART RXNE 中断 在后台逐字节接收：

void USART1_IRQHandler(void)
{
    if(USART_GetITStatus(USART1, USART_IT_RXNE) != RESET)
    {
        // 环形缓冲: 超限则回绕
        if(esp8266_cnt >= sizeof(esp8266_buf))
            esp8266_cnt = 0;

        esp8266_buf[esp8266_cnt++] = USART1->DR;

        USART_ClearFlag(USART1, USART_FLAG_RXNE);
    }
}
特性说明
缓冲区esp8266_buf[512] — 全局环形缓冲
写位置esp8266_cnt — ISR 中递增，主循环读取
回绕当 cnt ≥ 512 时重置为 0
中断触发USART_IT_RXNE — 每接收一个字节触发一次
为什么用中断？ ESP8266 的数据到达是完全异步且不可预测的。中断接收确保不会丢失任何字节，同时主循环可以专注于协议解析而不被串口接收阻塞。

4. AT 指令发送与响应等待

ESP8266_SendCmd 是模块最核心的通信函数：发送 AT 指令并等待期望的响应字符串。

_Bool ESP8266_SendCmd(char *cmd, char *res)
{
    unsigned char timeOut = 200;

    // 1. 发送 AT 指令字符串
    Usart_SendString(USART1, (unsigned char *)cmd,
                     strlen((const char *)cmd));

    // 2. 轮询等待响应, 超时 200 × 10ms = 2000ms
    while(timeOut--)
    {
        if(ESP8266_WaitRecive() == REV_OK)
        {
            // 3. 检查缓冲区是否包含期望字符串
            if(strstr((const char *)esp8266_buf, res) != NULL)
            {
                ESP8266_Clear();
                return 0;  // 成功
            }
        }
        DelayXms(10);
    }

    return 1;  // 超时失败
}

执行流程

  1. 通过 Usart_SendString 将 AT 指令发出（如 "AT\r\n"）
  2. 进入 200 次 轮询循环，每次等待 10 ms（共 2 秒超时）
  3. 每次轮询调用 ESP8266_WaitRecive() 检查数据是否接收完毕
  4. 若接收完毕，用 strstr 在缓冲区中搜索期望字符串（如 "OK"）
  5. 匹配成功则清缓冲区并返回 0，超时则返回 1
超时设计： 200 × 10 ms = 2 秒是普通 AT 指令的典型超时。对于 Wi-Fi 连接等耗时操作（如 AT+CWJAP），外部循环（while 重试）提供了额外的容错机制。

5. 初始化流程 — ESP8266_Init

模块初始化是一个 阻塞重试 流程：按顺序执行 5 个步骤，每步失败后会等待 500 ms 重试：

void ESP8266_Init(void)
{
    ESP8266_Clear();

    // 1. 测试 AT 通信
    while(ESP8266_SendCmd("AT\r\n", "OK"))
        DelayXms(500);

    // 2. 设置 Station 模式
    while(ESP8266_SendCmd("AT+CWMODE=1\r\n", "OK"))
        DelayXms(500);

    // 3. 启用 DHCP
    while(ESP8266_SendCmd("AT+CWDHCP=1,1\r\n", "OK"))
        DelayXms(500);

    // 4. 连接 Wi-Fi (等待 "GOT IP")
    while(ESP8266_SendCmd(ESP8266_WIFI_INFO, "GOT IP"))
        DelayXms(500);

    // 5. TCP 连接到 OneNet (等待 "CONNECT")
    while(ESP8266_SendCmd(ESP8266_ONENET_INFO, "CONNECT"))
        DelayXms(500);
}
步骤AT 指令期望响应说明
1ATOK测试 ESP8266 是否正常响应
2AT+CWMODE=1OK设置为 Station（客户端）模式
3AT+CWDHCP=1,1OK使能 Station DHCP 自动获取 IP
4AT+CWJAP="SSID","PASS"GOT IP连接 Wi-Fi 热点（耗时最长）
5AT+CIPSTART="TCP","..."CONNECT建立 TCP 连接到 OneNet MQTT

Wi-Fi 名称和密码定义在宏中：

#define ESP8266_WIFI_INFO    "AT+CWJAP=\"YOUR_WIFI_SSID\",\"YOUR_WIFI_PASSWORD\"\r\n"
#define ESP8266_ONENET_INFO  "AT+CIPSTART=\"TCP\",\"mqtts.heclouds.com\",1883\r\n"
注意： 实际使用时必须修改 ESP8266_WIFI_INFO 中的 SSID 和密码。OneNet 地址 mqtts.heclouds.com:1883 为中国移动物联网平台的标准 TCP 接入点。

6. 数据发送 — ESP8266_SendData

发送原始数据（如 MQTT 报文）需先通知 ESP8266 数据长度，等待 ">" 提示符后发送：

void ESP8266_SendData(unsigned char *data, unsigned short len)
{
    char cmdBuf[32];

    ESP8266_Clear();

    // 1. 发送 AT+CIPSEND=len
    sprintf(cmdBuf, "AT+CIPSEND=%d\r\n", len);

    // 2. 等待 ">" 提示符
    if(!ESP8266_SendCmd(cmdBuf, ">"))
    {
        // 3. 收到 ">" 后直接发送原始数据
        Usart_SendString(USART1, data, len);
    }
}
协议特点： ESP8266 的透传发送模式需要两步握手——先发 AT+CIPSEND=len 告知即将发送的字节数，等模组返回 ">" 提示符后再发送实际数据。如果直接发送数据，ESP8266 会将其当作 AT 命令解析。

7. 数据接收 — ESP8266_GetIPD

当 ESP8266 收到 TCP 数据时，会以 +IPD,<len>:<data> 格式推送。此函数解析该帧并返回数据指针：

unsigned char *ESP8266_GetIPD(unsigned short timeOut)
{
    char *ptrIPD = NULL;

    do
    {
        // 1. 等待数据接收完毕
        if(ESP8266_WaitRecive() == REV_OK)
        {
            // 2. 搜索 "+IPD," 或 "IPD," 前缀
            ptrIPD = strstr((char *)esp8266_buf, "+IPD,");
            if(ptrIPD == NULL)
                ptrIPD = strstr((char *)esp8266_buf, "IPD,");

            // 3. 找到冒号分隔符, 返回其后数据指针
            if(ptrIPD != NULL)
            {
                ptrIPD = strchr(ptrIPD, ':');
                if(ptrIPD != NULL)
                {
                    ptrIPD++;
                    return (unsigned char *)(ptrIPD);
                }
            }
        }

        DelayXms(5);
        timeOut--;
    } while(timeOut > 0);

    return NULL;  // 超时未收到 IPD 帧
}
解析示例： 如果缓冲区内容为 "\r\n+IPD,5:Hello\r\n"，函数返回指向 "Hello\r\n" 的指针。调用者需根据帧头中声明的长度（本例中为 5）读取对应字节数。

8. 环形缓冲区管理

模块基于 单生产者（ISR）— 单消费者（主循环） 模型管理接收数据：

unsigned char esp8266_buf[512];        // 接收缓冲区
unsigned short esp8266_cnt = 0;       // ISR 中的写入位置
unsigned short esp8266_cntPre = 0;    // 上一次快照位置

ESP8266_Clear — 清空缓冲

void ESP8266_Clear(void)
{
    memset(esp8266_buf, 0, sizeof(esp8266_buf));
    esp8266_cnt = 0;
}

ESP8266_WaitRecive — 等待数据稳定

_Bool ESP8266_WaitRecive(void)
{
    if(esp8266_cnt == 0)
        return REV_WAIT;     // 还没有数据

    if(esp8266_cnt == esp8266_cntPre)
    {
        // cnt 不再变化 = 数据接收完毕
        esp8266_cnt = 0;
        return REV_OK;       // 可以处理了
    }

    esp8266_cntPre = esp8266_cnt;  // 更新快照
    return REV_WAIT;            // 还在接收中
}
变量 / 函数作用
esp8266_cntISR 实时更新的写入位置
esp8266_cntPre保存上一次采样的 cnt 值
ESP8266_Clear()清空缓冲 + 重置 cnt
ESP8266_WaitRecive()返回 REV_OK 表示数据已稳定（不再增长）
工作原理： 当 ISR 向缓冲区写入数据时，cnt 不断递增。主循环每次调用 ESP8266_WaitRecive() 都会比较当前 cnt 与快照 cntPre。如果相等（且非零），说明 ISR 已停止写入，数据已完整到达。这种"静默检测"机制比固定延时更高效。

/* esp8266.c — ESP8266 驱动实现 */
#include "stm32f10x.h"
#include "esp8266.h"
#include "bsp_delay.h"
#include "bsp_usart.h"
#include <string.h>
#include <stdio.h>

#define ESP8266_WIFI_INFO   "AT+CWJAP=\"YOUR_WIFI_SSID\",\"YOUR_WIFI_PASSWORD\"\r\n"
#define ESP8266_ONENET_INFO "AT+CIPSTART=\"TCP\",\"mqtts.heclouds.com\",1883\r\n"

unsigned char esp8266_buf[512];
unsigned short esp8266_cnt = 0, esp8266_cntPre = 0;

// 清空接收缓冲区
void ESP8266_Clear(void)
{
    memset(esp8266_buf, 0, sizeof(esp8266_buf));
    esp8266_cnt = 0;
}

// 等待数据稳定 (不再接收新字节)
_Bool ESP8266_WaitRecive(void)
{
    if(esp8266_cnt == 0)
        return REV_WAIT;

    if(esp8266_cnt == esp8266_cntPre)
    {
        esp8266_cnt = 0;
        return REV_OK;
    }

    esp8266_cntPre = esp8266_cnt;
    return REV_WAIT;
}

// 发送 AT 指令并等待期望响应
_Bool ESP8266_SendCmd(char *cmd, char *res)
{
    unsigned char timeOut = 200;

    Usart_SendString(USART1,
        (unsigned char *)cmd, strlen((const char *)cmd));

    while(timeOut--)
    {
        if(ESP8266_WaitRecive() == REV_OK)
        {
            if(strstr((const char *)esp8266_buf, res) != NULL)
            {
                ESP8266_Clear();
                return 0;  // 成功
            }
        }
        DelayXms(10);
    }
    return 1;  // 超时
}

// 发送原始数据 (MQTT 报文等)
void ESP8266_SendData(unsigned char *data, unsigned short len)
{
    char cmdBuf[32];
    ESP8266_Clear();
    sprintf(cmdBuf, "AT+CIPSEND=%d\r\n", len);
    if(!ESP8266_SendCmd(cmdBuf, ">"))
    {
        Usart_SendString(USART1, data, len);
    }
}

// 解析 +IPD 帧, 返回数据指针
unsigned char *ESP8266_GetIPD(unsigned short timeOut)
{
    char *ptrIPD = NULL;

    do
    {
        if(ESP8266_WaitRecive() == REV_OK)
        {
            ptrIPD = strstr((char *)esp8266_buf, "+IPD,");
            if(ptrIPD == NULL)
                ptrIPD = strstr((char *)esp8266_buf, "IPD,");

            if(ptrIPD != NULL)
            {
                ptrIPD = strchr(ptrIPD, ':');
                if(ptrIPD != NULL)
                {
                    ptrIPD++;
                    return (unsigned char *)(ptrIPD);
                }
            }
        }
        DelayXms(5);
        timeOut--;
    } while(timeOut > 0);

    return NULL;  // 超时
}

// 初始化: AT → 模式 → DHCP → WiFi → TCP
void ESP8266_Init(void)
{
    ESP8266_Clear();

    while(ESP8266_SendCmd("AT\r\n", "OK"))
        DelayXms(500);

    while(ESP8266_SendCmd("AT+CWMODE=1\r\n", "OK"))
        DelayXms(500);

    while(ESP8266_SendCmd("AT+CWDHCP=1,1\r\n", "OK"))
        DelayXms(500);

    while(ESP8266_SendCmd(ESP8266_WIFI_INFO, "GOT IP"))
        DelayXms(500);

    while(ESP8266_SendCmd(ESP8266_ONENET_INFO, "CONNECT"))
        DelayXms(500);
}

// USART1 中断: RXNE 逐字节存入环形缓冲
void USART1_IRQHandler(void)
{
    if(USART_GetITStatus(USART1, USART_IT_RXNE) != RESET)
    {
        if(esp8266_cnt >= sizeof(esp8266_buf))
            esp8266_cnt = 0;

        esp8266_buf[esp8266_cnt++] = USART1->DR;
        USART_ClearFlag(USART1, USART_FLAG_RXNE);
    }
}
